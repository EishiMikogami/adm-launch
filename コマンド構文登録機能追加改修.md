function _cleanup_on_exit() {
  # 1. メモリ上の機密変数を完全に上書きして破棄
  master_key="00000000000000000000"
  user_pass="00000000000000000000"
  decrypted_data="00000000000000000000"
  totp_secret="00000000000000000000"
  input_otp="000000"
  correct_otp="000000"

trap _cleanup_on_exit EXIT SIGINT SIGTERM

# --- 関数: セキュア抹消 ---
function _secure_cleanup() {
  local target="$1"
  [ ! -f "$target" ] && return

  if command -v shred &>/dev/null; then
    shred -u -n 1 "$target" 2>/dev/null
  else
    local size
    size=$(wc -c <"$target" | tr -d ' ')
    [ "$size" -gt 0 ] && dd if=/dev/urandom of="$target" bs=1 count="$size" conv=notrunc &>/dev/null
    rm -f "$target"
  fi
}

# --- 関数: ユーザー認証 ---
function authenticate_admin() {
  # 1. 必要ツールの確認
  if ! command -v openssl &>/dev/null; then
    echo "Error: openssl is required." >&2
    return 1
  fi

  # 2. マスターキー入力と復号 (オンメモリ)
  read -sp "Enter Master Key: " master_key
  echo

  decrypted_data=$(openssl enc -d -aes-256-cbc -pbkdf2 -iter 100000 -k "$master_key" -in "$ENC_DB" 2>/dev/null)

  if [ $? -ne 0 ] || [ -z "$decrypted_data" ]; then
    echo "Error: Decryption failed." >&2
    _log_event "FAILURE" "Decryption failed (Invalid Master Key)"
    return 1
  fi

  stored_hash=$(echo "$decrypted_data" | cut -d',' -f1)
  totp_secret=$(echo "$decrypted_data" | cut -d',' -f2)

  # 元データはここで一旦上書きしてから消去（よりセキュアに）
  decrypted_data="0000000000"
  unset decrypted_data

  # 3. 本人認証パスワードの照合
  read -sp "Enter Admin Password: " user_pass
  echo
  # input_hash=$(echo -n "${user_pass}${SALT}" | openssl dgst -sha256 | awk '{print $2}')
  # openssl dgst -sha256 -r を使うと、常に「ハッシュ値 [スペース] *stdin」の形式になるため、確実に1番目を抜き出せます
  input_hash=$(echo -n "${user_pass}${SALT}" | openssl dgst -sha256 -r | awk '{print $1}')

  # --- デバッグ用の一時挿入 ---
  # echo "--- DEBUG ---"
  # printf "Stored Hash (HEX): %s\n" "$(echo -n "$stored_hash" | xxd -p)"
  # printf "Input  Hash (HEX): %s\n" "$(echo -n "$input_hash" | xxd -p)"
  # echo "--------------"

  if [[ "$input_hash" != "$stored_hash" ]]; then
    echo "Error: Authentication failed." >&2
    _log_event "FAILURE" "Admin password mismatch"
    return 1
  fi

  # 4. 2要素認証 (TOTP) - oathtool導入確認

  if [[ -n "$totp_secret" ]] && ! command -v oathtool &>/dev/null; then
    echo "Error: totp_secret が設定されていますが、oathtool がインストールされていません。" >&2
    _log_event "FAILURE" "Not Installed oathtool"
    return 1
  fi

  # 4. 3要素認証 (TOTP) - オプション
  if command -v oathtool &>/dev/null && [[ -n "$totp_secret" ]]; then
    read -p "Enter OTP: " input_otp
    correct_otp=$(oathtool --totp -b "$totp_secret")

    if [[ "$input_otp" != "$correct_otp" ]]; then
      echo "Error: Invalid OTP." >&2
      _log_event "FAILURE" "Invalid OTP"
      return 1
    fi
  fi

  # 5. 後始末
  _log_event "SUCCESS" "Authentication successful"
  echo "Authentication Success."
  return 0
}

# ログ記録用関数
function _log_event() {
  local status="$1"
  local message="$2"

    touch "$LOG_FILE"
    chmod 600 "$LOG_FILE"
  fi

  printf "[%s] [%s] %s\n" "$(date '+%Y-%m-%d %H:%M:%S')" "$status" "$message" >>"$LOG_FILE"
}

# このコードに改修を行いたいと考えます。 内容として,は現行のシェルスクリプトを登録する機能に加え、コマンド構文を登録する機能も実装する。　また、RASISの観点から通常のターミナルからコマンドを叩くときより、安全性を下げないもしくは、より高いものとする。 登録したものへヘッダを付加する（シェルスクリプトの場合は「shell:」,コマンドの場合は「cmd:」）。　実行ログとして、登録名を記録する。

このような内容の改修に最適な、コードの提案をお願いしたいのですが、可能でしょうか？

ご提示いただいたコードは、オンメモリでの機密情報の上書きや `trap` による確実なクリーンアップなど、セキュリティに非常に深く配慮された素晴らしい設計になっています。

今回の「シェルスクリプトに加えコマンド構文も登録可能にする」「ヘッダ（`shell:`, `cmd:`）による識別」「RASIS（特に信頼性・保守性・安全性）の強化」という要件を満たす改修コードを提案いたします。

---

## 💡 RASISの観点からの強化ポイント

1. **安全性 (Security) & 信頼性 (Reliability)**

- **不正なコマンド実行（インジェクション）の防止:** コマンド構文（`cmd:`）を実行する際、不用意な評価による脆弱性を防ぐため、安易な `eval` を避け、実行前に最低限の空文字チェックやサニタイズを考慮できる構造にしています。
- **配列インデックスの詰め直しバグ修正:** 元のコードの削除（`d`）処理にあった「`unset` 後にインデックスが不連続（虫食い）になり、ループや要素数カウントで意図しない挙動になる」バグを、`load_config` 時に配列を再生成することで完全に解決しました（信頼性の向上）。

2. **保守性 (Maintainability) & 可用性 (Availability)**

- **ヘッダ識別子による透過的な処理:** 設定ファイル（`$CONFIG_FILE_LAUNCH_ADMIN`）内に `shell:` / `cmd:` のヘッダを付加して保存し、読み込み時に自動で種別を判定します。

## 🛠 主な変更点と解説

### 1. `paths` 配列から `targets` 配列への名称変更と汎用化

従来のスクリプトパス専用配列 `paths` を、コマンドも格納できる `targets` に変更しました。データ構造として `shell:/path/to/script.sh` または `cmd:echo "Hello"` の形式で保存・管理します。

### 2. 安全性の高いコマンド実行 (`/usr/bin/env bash -c`)

コマンド構文を実行する際、現行プロセスのシェル環境を汚染・破壊されるリスク（変数の予期せぬ上書きなど）を抑えるため、独立した環境の `/usr/bin/env bash -c "$body"` を介して安全に評価・実行するようにしています。

### 3. 実行ログ（監査トレイル）の強化

要求のあった「登録名（タイトル）」をログに残す機能を組み込みました。

```bash
_log_event "EXEC_SHELL" "Name: $item_title, Path: $body"
_log_event "EXEC_CMD" "Name: $item_title, Cmd: $body"

```

これにより、後から「誰が・いつ・どの登録名の処理を実行したか」が完全に追跡可能になり、RASISにおける **安全性 (Security)** と **信頼性 (Reliability)** をより強固にしています。

## 1. テストフェーズ（推奨される流れ）

シェルスクリプトのテストは、通常のアプリケーション開発と同様に以下の3ステップで進めるのが最も安全です。

```
[Phase 1: 静的解析] ──> [Phase 2: 単体・結合テスト] ──> [Phase 3: 運用・破壊テスト]
  (コードを動かさない)       (安全な隔離環境で実行)         (例外系や不正入力をテスト)

```

### Phase 1: 静的解析（コードを動かさない検証）

- **目的:** 構文エラーや、シェルスクリプト特有の一番危ないバグ（スペースの有無による誤挙動など）を事前に100%潰します。
- **作法:** 静的解析ツール（Linter）を通します。

### Phase 2: 単体・結合テスト（隔離環境での機能検証）

- **目的:** 認証の成功/失敗、シェルスクリプト/コマンド構文の登録・実行が意図通り動くか確認します。
- **作法:** 本番環境（`~/.passcode.enc` など）を絶対に汚さないよう、**テスト用のダミー環境**をスクリプト内で一時的に指定して動かします。

### Phase 3: 運用・破壊テスト（安全性の検証）

- **目的:** 不正な入力や、途中で強制終了（Ctrl+C）された際にも機密情報が残らないか、RASISの「安全性・保全性」を検証します。

---

## 2. テストを安全・確実に行うための「作法」

### ① テスト用環境の分離（最重要）

テストを行う際は、スクリプト冒頭の設定（パス設定）を以下のように**テスト専用のパス**に書き換えて実施してください。本番のパスワードデータベースを誤って上書き・削除するリスクを防ぎます。

```bash
# テスト時は一時的に一時ディレクトリなどに向ける
ENC_DB=~/".test_passcode.enc"
LOG_FILE=~/".test_usr_access.log"
CONFIG_FILE_LAUNCH_ADMIN=~/".test_adm_launcher_config.txt"

```

### ② 静的解析ツールの導入

シェルスクリプトのバグを自動検知する **`ShellCheck`** というツールの利用を強く推奨します。

- **実行方法:** ターミナルで `shellcheck your_script.sh` を実行するだけです。
- これにより、クォーテーションの囲み忘れや、未定義変数の参照など、セキュリティホールになり得る箇所を漏れなく指摘してくれます。

## 3. テスト項目チェックリスト

テスト時に「何を検証すべきか」を網羅したチェックリストです。

### 📋 1. 認証・セキュリティ（S: 安全性・保全性）

- [ ] **正しいキーでの復号:** 正しいマスターキー・パスワード・OTPを入力した際、正常にランチャーメニューへ移行するか。
- [ ] **誤ったキーの拒否:** マスターキー、パスワード、OTPのいずれかが1文字でも違っていた場合、確実に処理がブロックされるか。
- [ ] **OTPの必須チェック:** `totp_secret` が暗号データ内にあるにもかかわらず `oathtool` がシステムにない場合、安全にエラー終了するか。
- [ ] **オンメモリの抹消（トラップ検知）:** 認証成功後、およびスクリプト実行中に `Ctrl+C` で強制終了した際、`master_key` などの変数が残っていないか（スクリプト内に `echo $master_key` などを一時的に仕込んで検証）。

### 📋 2. ランチャー機能（R: 信頼性・A: 可用性）

- [ ] **ヘッダの自動付与:** 新規登録（`a`）でシェル（1）を選んだとき `shell:`、コマンド（2）を選んだとき `cmd:` が設定ファイルに正しく書き込まれるか。
- [ ] **シェルスクリプトの実行:** 登録したシェルスクリプトが正常に呼び出され、終了後にEnterキーでメニューに戻るか。
- [ ] **コマンド構文の実行:** `echo "Hello"` や `ls -la` などのコマンド構文が正常に実行されるか。
- [ ] **存在しないファイルのハンドリング:** `shell:` で登録したファイルが削除されていた場合、システムがクラッシュせず「ファイルが見つかりません」とエラーを表示してメニューに戻るか。

### 📋 3. データ操作・配列（M: 保守性）

- [ ] **削除時の虫食い防止:** 複数（例: 3つ）登録した状態から真ん中の番号（2番）を削除（`d`）した後、メニューの番号が「1, 2」と綺麗に詰め直され、ズレなく次の操作ができるか。
- [ ] **入れ替え（ソート）:** `s` を選択し、要素の入れ替えが設定ファイル（`.test_adm_launcher_config.txt`）側にも正しく同期されるか。
- [ ] **上限値の制限:** 登録数が `LIMIT=50` に達した際、新規登録が拒否されるか。

### 📋 4. 監査ログ（監査可能性）

- [ ] **成否ログの記録:** 認証成功時（`SUCCESS`）、失敗時（`FAILURE`）のログが日時とともに正確に記録されているか。
- [ ] **実行ログの記録:** シェルやコマンドを実行した際、「登録名（タイトル）」と「実行内容」がログファイルに正しく追跡記録されているか。
- [ ] **ログの権限:** 生成されたログファイルのパーミッションが `600`（所有者のみ読み書き可能）になっているか。
